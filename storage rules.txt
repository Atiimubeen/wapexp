rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Is rule ka matlab hai ke Storage ke andar mojood har file ko...
    match /{allPaths=**} {
      // Koi bhi parh (read) sakta hai
      allow read;
      // Sirf login user hi likh (write/upload/delete) sakta hai
      allow write: if request.auth != null;
    }
  }
}
