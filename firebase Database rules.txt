rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // =================================
    // WAPEXP APP RULES
    // =================================

    match /users/{userId} {
      allow read: if request.auth != null;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null;
    }

    match /courses/{courseId} {
      allow read;
      allow write: if request.auth != null;
    }

    match /achievements/{achievementId} {
      allow read;
      allow write: if request.auth != null;
    }

    match /sessions/{sessionId} {
      allow read;
      allow write: if request.auth != null;
    }

    match /sliderImages/{imageId} {
      allow read;
      allow write: if request.auth != null;
    }

    match /announcements/{announcementId} {
      allow read;
      allow write: if request.auth != null;
    }

    // =================================
    // AAPKE DOSTON KE PURANE RULES (UNCHANGED)
    // =================================
    match /properties/{propertyId} {
      allow read;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.landlordId;
    }

    match /bookings/{bookingId} {
      allow read: if request.auth != null && (request.auth.uid == resource.data.tenantId || request.auth.uid == resource.data.landlordId);
      allow create: if request.auth != null && request.resource.data.tenantId == request.auth.uid;
      allow update: if request.auth != null && request.auth.uid == resource.data.landlordId;
      allow delete: if request.auth != null && request.auth.uid == resource.data.tenantId;
    }

    match /chat_rooms/{chatRoomId} {
      allow get: if request.auth != null && request.auth.uid in resource.data.userIds;
      allow write: if request.auth != null && request.auth.uid in request.resource.data.userIds;

      match /messages/{messageId} {
        allow read: if request.auth != null && get(/databases/$(database)/documents/chat_rooms/$(chatRoomId)).data.userIds.hasAny([request.auth.uid]);
        allow create: if request.auth != null && request.auth.uid == request.resource.data.senderId;
        allow delete: if request.auth != null && request.auth.uid == resource.data.senderId;
      }
    }
  }
}
